Capstone Project 3 — Consolidation

Mini ERP Console Application

This is your first complete application that combines all 9 modules into one cohesive program. It is a simplified ERP with a student roster, course catalog, enrollment management, fee calculation, grade tracking, event notifications, and a LINQ-powered analytics report — all driven by an interactive menu loop.

⏱ Duration: ~2 hours 📦 Modules: 1–9 📁 File: Capstone3.cs

// ── Enums ────────────────────────────────── enum ErpStatus { Active, Inactive, Suspended } // ── Student model ────────────────────────── class ErpStudent { public string Id { get; } public string Name { get; set; } public int Age { get; set; } public ErpStatus Status { get; set; } public int Marks { get; set; } public string EnrolledCourseCode { get; set; } public ErpStudent(string id, string name, int age) { Id = id; Name = name; Age = age; Status = ErpStatus.Active; Marks = 0; EnrolledCourseCode = ""; } public string GetGrade() { if (Marks >= 90) return "A"; if (Marks >= 75) return "B"; if (Marks >= 50) return "C"; return "F"; } } // ── Course model ─────────────────────────── class ErpCourse { public string Code { get; } public string Title { get; set; } public int DurationDays { get; set; } public double Price { get; set; } public ErpCourse(string code, string title, int days, double price) { Code = code; Title = title; DurationDays = days; Price = price; } public double PriceWithGST() => Price + (Price * 0.18); }
PHASE 2

Repositories — in-memory data stores

M7 — Dictionary + List
1 Add two simple repository classes

A repository is just a class that holds and manages a collection. It gives you clean methods like Add, GetById, GetAll — instead of accessing the raw Dictionary everywhere.

StudentRepo and CourseRepo — bottom of file
class StudentRepo { private Dictionary<string, ErpStudent> _students = new Dictionary<string, ErpStudent>(); public void Add(ErpStudent s) => _students[s.Id] = s; public ErpStudent GetById(string id) { _students.TryGetValue(id, out ErpStudent s); return s; // null if not found } public List<ErpStudent> GetAll() => _students.Values.ToList(); } class CourseRepo { private Dictionary<string, ErpCourse> _courses = new Dictionary<string, ErpCourse>(); public void Add(ErpCourse c) => _courses[c.Code] = c; public ErpCourse GetByCode(string code) { _courses.TryGetValue(code, out ErpCourse c); return c; } public List<ErpCourse> GetAll() => _courses.Values.ToList(); }

class AlreadyEnrolledException : Exception { public AlreadyEnrolledException(string msg) : base(msg) { } } class EnrollmentService { public event Action<ErpStudent, ErpCourse> OnEnrolled; public void Enroll(ErpStudent student, ErpCourse course) { if (student == null) throw new Exception("Student not found"); if (course == null) throw new Exception("Course not found"); if (student.EnrolledCourseCode != "") throw new AlreadyEnrolledException( student.Name + " is already enrolled in " + student.EnrolledCourseCode); student.EnrolledCourseCode = course.Code; OnEnrolled?.Invoke(student, course); } } // GradeService (same as Capstone 2) class ErpGradeService { public event Action<string, int> OnGradePosted; public void PostGrade(ErpStudent student, int marks) { if (student == null) throw new Exception("Student not found"); student.Marks = marks; OnGradePosted?.Invoke(student.Name, marks); } }

1 Interface + two report implementations

interface IErpReport { void Print(List<ErpStudent> students, List<ErpCourse> courses); } class EnrollmentReport : IErpReport { public void Print(List<ErpStudent> students, List<ErpCourse> courses) { Console.WriteLine("--- Enrollment Report ---"); var enrolled = students.Where(s => s.EnrolledCourseCode != ""); foreach (var s in enrolled) { var course = courses.FirstOrDefault(c => c.Code == s.EnrolledCourseCode); string courseTitle = course != null ? course.Title : "Unknown"; Console.WriteLine(s.Id + " | " + s.Name.PadRight(8) + " | " + s.EnrolledCourseCode + " | " + courseTitle.PadRight(20) + " | " + s.Status); } } } class AnalyticsReport : IErpReport { public void Print(List<ErpStudent> students, List<ErpCourse> courses) { Console.WriteLine("--- Analytics Report ---"); var graded = students.Where(s => s.Marks > 0).ToList(); if (graded.Count == 0) { Console.WriteLine("No grades posted yet."); return; } Console.WriteLine("Total graded : " + graded.Count); Console.WriteLine("Average marks : " + graded.Average(s => s.Marks).ToString("F1")); var top = graded.OrderByDescending(s => s.Marks).First(); Console.WriteLine("Top student : " + top.Name + " (" + top.Marks + ")"); int passRate = graded.Count(s => s.Marks >= 50) * 100 / graded.Count; Console.WriteLine("Pass rate : " + passRate + "%"); Console.Write("Top scorers : "); foreach (var s in graded.Where(s => s.Marks >= 75).OrderByDescending(s => s.Marks)) Console.Write(s.Name + " (" + s.Marks + ") "); Console.WriteLine(); } }

static void ShowMenu() { Console.WriteLine("\n=== MAIN MENU ==="); Console.WriteLine("1. Show Students"); Console.WriteLine("2. Show Courses"); Console.WriteLine("3. Enroll Student"); Console.WriteLine("4. Post Grade"); Console.WriteLine("5. Run Reports"); Console.WriteLine("0. Exit"); Console.Write("Choice: "); } static void Log(string level, string message) { Console.WriteLine("[" + level + "] " + DateTime.Now.ToString("HH:mm:ss") + " - " + message); } static void ShowStudents(StudentRepo repo) { var list = repo.GetAll(); if (list.Count == 0) { Console.WriteLine("No students."); return; } foreach (var s in list) Console.WriteLine(s.Id + " | " + s.Name + " | Age: " + s.Age + " | " + s.Status); } static void ShowCourses(CourseRepo repo) { foreach (var c in repo.GetAll()) Console.WriteLine(c.Code + " | " + c.Title + " | " + c.DurationDays + " days | ₹" + c.PriceWithGST().ToString("F0") + " (incl. GST)"); } static void HandleEnroll(StudentRepo sRepo, CourseRepo cRepo, EnrollmentService svc) { Console.Write("Student ID: "); string sid = Console.ReadLine() ?? ""; Console.Write("Course Code: "); string cid = Console.ReadLine() ?? ""; try { Log("INFO", "Enrolling " + sid + " in " + cid); svc.Enroll(sRepo.GetById(sid), cRepo.GetByCode(cid)); } catch (AlreadyEnrolledException ex) { Log("ERROR", ex.Message); } catch (Exception ex) { Log("ERROR", ex.Message); } } static void HandleGrade(StudentRepo repo, ErpGradeService svc) { Console.Write("Student ID: "); string sid = Console.ReadLine() ?? ""; Console.Write("Marks: "); bool ok = int.TryParse(Console.ReadLine(), out int marks); if (!ok) { Console.WriteLine("Invalid marks."); return; } try { svc.PostGrade(repo.GetById(sid), marks); } catch (Exception ex) { Log("ERROR", ex.Message); } }

Wire Everything Together in Run()

All modules
1 The complete Run() method
Complete Run() — the entire entry point
public static void Run() { Console.WriteLine("========================================="); Console.WriteLine(" Mini ERP — RR Skillverse"); Console.WriteLine(" Intelligent Enterprise Development"); Console.WriteLine("========================================="); // Seed data StudentRepo studentRepo = new StudentRepo(); studentRepo.Add(new ErpStudent("S001", "Raushan", 28)); studentRepo.Add(new ErpStudent("S002", "Priya", 24)); studentRepo.Add(new ErpStudent("S003", "Rahul", 26)); CourseRepo courseRepo = new CourseRepo(); courseRepo.Add(new ErpCourse("CS-001", "C# Foundations", 3, 14999)); courseRepo.Add(new ErpCourse("AZ-204", "Azure Developer", 5, 24999)); courseRepo.Add(new ErpCourse("AI-001", "Azure AI Fundamentals", 11, 29999)); // Services + event subscriptions EnrollmentService enrollSvc = new EnrollmentService(); enrollSvc.OnEnrolled += (s, c) => Console.WriteLine("[EVENT] Email: " + s.Name + " enrolled in " + c.Title); enrollSvc.OnEnrolled += (s, c) => Console.WriteLine("[EVENT] Log : Enrollment recorded"); ErpGradeService gradeSvc = new ErpGradeService(); gradeSvc.OnGradePosted += (name, marks) => { string g = marks >= 90 ? "A" : marks >= 75 ? "B" : marks >= 50 ? "C" : "F"; Console.WriteLine("[EVENT] Email: " + name + " scored " + marks + " (Grade: " + g + ")"); }; gradeSvc.OnGradePosted += (name, marks) => Console.WriteLine("[EVENT] Log : Grade posted"); // Reports List<IErpReport> reports = new List<IErpReport> { new EnrollmentReport(), new AnalyticsReport() }; // Main loop string choice = ""; while (choice != "0") { ShowMenu(); choice = Console.ReadLine() ?? "0"; switch (choice) { case "1": ShowStudents(studentRepo); break; case "2": ShowCourses(courseRepo); break; case "3": HandleEnroll(studentRepo, courseRepo, enrollSvc); break; case "4": HandleGrade(studentRepo, gradeSvc); break; case "5": var allStudents = studentRepo.GetAll(); var allCourses = courseRepo.GetAll(); foreach (var r in reports) { Console.WriteLine(); r.Print(allStudents, allCourses); } break; case "0": Console.WriteLine("Goodbye!"); break; default: Console.WriteLine("Invalid choice. Try again."); break; } } }
💡

Lambda expressions (s, c) => Console.WriteLine(...) are used to subscribe event handlers inline. This is the same as writing a separate static method and subscribing it — just more concise when the handler is short.
